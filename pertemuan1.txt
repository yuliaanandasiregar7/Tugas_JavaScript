//proses DDL Create
//membuat table
CREATE TABLE jenis(
	id int PRIMARY KEY AUTO_INCREMENT,
    nama varchar(30) NOT NULL UNIQUE
);
/* penjelasan : id adalah nama kolom atau header kolom
             primary key adalah key atau kunci dari sebuah table
             AUTO_INCREMENT adalah key untuk autoincrement penomoran
             NOT NULL Menandakan kolom tersebut tidak boleh kosong jika data mulai diisi
             UNIQUE adalah key untuk kolom agar data tidak boleh sama */

//melihat hasil table jenis
DESC jenis; 

//membuat relasi antara table jenis dan table produk
CREATE TABLE produk(
	id int PRIMARY KEY AUTO_INCREMENT,
    nama varchar(30) NOT NULL UNIQUE,
    harga double NOT NULL,
   	stok int NOT NULL,
    idjenis int NOT NULL REFERENCES jenis(id),
    foto varchar(30)
);
